



                    ASSIGNMENT ON LOGICAL OPERATOR.


1. WRITE A QUERY TO DISPLAY DETAILS OF THE EMPLOYEES WORKING AS CLERK AND EARNING LESS THAN 1500.
    ==> select * from emp where job = 'CLERK' AND sal < 1500 ;

2. WRITE A QUERY TO DISPLAY NAME AND HIREDATE OF THE EMPLOYEES WORKING AS MANAGER IN DEPT 30.
    ==> select ename, hiredate, deptno from emp where job = 'MANAGER' AND deptno=30 ;

3. WRITE A QUERY TO DISPLAY DETAILS OF THE EMP ALONG WITH ANNUAL SALARY IF THEY ARE WORKING IN DEPT 30 AS SALESMAN AND THEIR ANNUAL SALARY HAS TO BE GREATER THAN 14000.
    ==> select emp.*, sal*12 from emp  where deptno = 30 AND job = 'SALESMAN' AND sal*12 > 14000 ;

4. WRITE A QUERY TO DISPLAY ALL THE DETAILS OF THE EMP WORKING IN DEPT 30 OR AS ANALYST.
    ==> select * from emp where deptno = 30 AND job = 'ANALYST' ;

5. WRITE A QUERY TO DISPLAY NAMES OF THE EMPMLOYEES WHOS SALARY IS LESS THAN 1100 AND THEIR DESIGNATION IS CLERK.
    ==> select ename from emp where sal < 1100 AND job = 'CLERK' ;

6. WRITE A QUERY TO DISPLAY NAME AND SAL , ANNUAL SAL AND DEPTNO IF DEPTNO IS 20 EARNING MORE THAN 1100 AND ANNUAL SALARY EXCEEDS 12000.
    ==> select ename, sal, sal*12 from emp where deptno= 20  AND sal > 1100 AND sal*12 > 12000 ;

7.  WRITE A QUERY TO DISPLAY EMPNO AND NAMES OF THE EMPLOYEES WORKING AS MANAGER IN DEPT 20.
    ==> select empno, ename, deptno, job from emp where job = 'MANAGER' AND deptno = 20 ;

8. WRITE A QUERY TO DISPLAY DETAILS OF EMPLOYEES WORKING IN DEPT 20 OR 30 .
    ==> select * from emp where deptno IN (20,30);
            OR
        select * from emp where deptno = 20 OR deptno=30 ;

9. WRITE A QUERY TO DISPLAY DETAILS OF EMPLOYEES WORKING AS ANALYST IN DEPT 10 .

10. WRITE A QUERY TO DISPLAY DETAILS OF EMPLOYEE WORKING AS PRESIDENT WITH SALARY OF RUPEES 4000.

11. WRITE A QUERY TO DISPLAY NAMES AND DEPTNO , JOB OF EMPS WORKING AS CLERK IN DEPT 10 OR 20.
    ==> select ename, deptno from emp where job= 'CLERK' AND deptno IN (10,20) ;

12. WRITE A QUERY TO DISPLAY DETAILS OF EMPLOYEES WORKING AS CLERK OR MANAGER IN DEPT 10 .
    ==> select * from emp where job IN ('CLERK','MANAGER') AND deptno = 10;

13. WRITE A QUERY TO DISPLAY AMES OF EMPLOYEES WORKING IN DEPT 10 , 20 , 30 , 40 .

                OR
        select ename from emp where deptno = 10 OR deptno = 20 OR deptno = 30 OR deptno = 40 ;

18. WRITE A QUERY TO DISPLAY NAMES OF EMPLOYEES HIRED AFTER 81 INTO DEPT 10 OR 30.
